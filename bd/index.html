Script TP 1 — tp1.sql
-- ========================================== -- TP-1: MySQL Database Implementation -- ========================================== CREATE DATABASE IF NOT EXISTS GestionSalles; USE GestionSalles; -- 1. Create Departement Table CREATE TABLE departement ( departement_id INTEGER PRIMARY KEY, nom_departement VARCHAR(25) UNIQUE NOT NULL ); -- 2. Create Salle Table CREATE TABLE salle ( batiment VARCHAR(1), numero_salle VARCHAR(10), capacite INTEGER CHECK (capacite > 10), PRIMARY KEY (batiment, numero_salle) ); -- 3. Create Enseignement Table CREATE TABLE enseignement ( enseignement_id INTEGER, departement_id INTEGER, intitule VARCHAR(60), description VARCHAR(1000), PRIMARY KEY (enseignement_id, departement_id), FOREIGN KEY (departement_id) REFERENCES departement(departement_id) ); -- 4. Create Enseignant Table CREATE TABLE enseignant ( enseignant_id INTEGER PRIMARY KEY, departement_id INTEGER, nom VARCHAR(25), prenom VARCHAR(25), grade VARCHAR(25) CHECK (grade IN ('Vacataire', 'Moniteur', 'ATER', 'MCF', 'PROF')), telephone VARCHAR(10), fax VARCHAR(10), email VARCHAR(100), FOREIGN KEY (departement_id) REFERENCES departement(departement_id) ); -- 5. Create Etudiant Table CREATE TABLE Etudiant ( Etudiant_ID INTEGER PRIMARY KEY, Nom VARCHAR(25) NOT NULL, Prenom VARCHAR(25) NOT NULL, Date_Naissance DATE NOT NULL, Adresse VARCHAR(50) DEFAULT NULL, Ville VARCHAR(25) DEFAULT NULL, Code_Postal VARCHAR(9) DEFAULT NULL, Telephone VARCHAR(10) DEFAULT NULL, Fax VARCHAR(10) DEFAULT NULL, Email VARCHAR(100) DEFAULT NULL ); -- 6. Create Reservation Table CREATE TABLE reservation ( reservation_id INTEGER PRIMARY KEY, batiment VARCHAR(1) NOT NULL, numero_salle VARCHAR(10) NOT NULL, enseignement_id INTEGER NOT NULL, departement_id INTEGER NOT NULL, enseignant_id INTEGER NOT NULL, date_resa DATE NOT NULL DEFAULT (CURRENT_DATE), heure_debut TIME NOT NULL DEFAULT (CURRENT_TIME), heure_fin TIME NOT NULL DEFAULT '23:00:00', nombre_heures INTEGER NOT NULL CHECK (nombre_heures >= 1), CONSTRAINT FK_Reservation_Salle FOREIGN KEY (batiment, numero_salle) REFERENCES salle(batiment, numero_salle) ON UPDATE RESTRICT ON DELETE RESTRICT, CONSTRAINT FK_Reservation_Enseignement FOREIGN KEY (enseignement_id, departement_id) REFERENCES enseignement(enseignement_id, departement_id) ON UPDATE RESTRICT ON DELETE RESTRICT, CONSTRAINT FK_Reservation_Enseignant FOREIGN KEY (enseignant_id) REFERENCES enseignant(enseignant_id) ON UPDATE RESTRICT ON DELETE RESTRICT, CONSTRAINT CK_Reservation_HeureDebFin CHECK (heure_debut < heure_fin) ); -- 7. Create View CREATE OR REPLACE VIEW Email_Etudiant AS SELECT Nom, Prenom, Email FROM Etudiant; -- INSERTIONS INSERT INTO departement VALUES (1, 'IRT'), (2, 'IGL'), (3, 'ILA'); INSERT INTO salle VALUES ('B', '020', 15), ('B', '022', 15), ('A', '301', 45), ('C', 'Amphi 8', 500); INSERT INTO enseignant VALUES (1, 1, 'ousteith', 'ouahed', 'MCF', '4185', '4091', 'ousteith@gmail.com'); INSERT INTO Etudiant VALUES (1, 'ben foulen', 'foulen', '1979-02-18', '50, Rue des alouettes', 'TUNIS', '75021', '0143567890', NULL, 'foulen@gmail.com'); INSERT INTO enseignement VALUES (1, 1, 'Bases de Données Relationnelles', 'Modélisation E/A et SQL'); INSERT INTO reservation VALUES (1, 'B', '022', 1, 1, 1, '2008-10-15', '08:30:00', '11:45:00', 3);
Script TP 2 — tp2.sql
USE GestionSalles; -- 1. Liste des noms et des prénoms des étudiants SELECT Nom, Prenom FROM Etudiant; -- 2. Étudiants habitant Tunis, puis Sfax SELECT Nom, Prenom FROM Etudiant WHERE Ville = 'TUNIS'; SELECT Nom, Prenom FROM Etudiant WHERE Ville = 'Sfax'; -- 3. Étudiants dont le nom commence par 't', puis par 'f' SELECT Nom, Prenom FROM Etudiant WHERE Nom LIKE 't%'; SELECT Nom, Prenom FROM Etudiant WHERE Nom LIKE 'f%'; -- 4. Enseignants dont l'avant-dernière lettre du nom est 'a' SELECT Nom, Prenom FROM Enseignant WHERE Nom LIKE '%a_'; -- 5. Enseignants classés par département, nom et prénom SELECT E.Nom, E.Prenom, D.nom_departement FROM Enseignant E JOIN departement D ON E.departement_id = D.departement_id ORDER BY D.nom_departement ASC, E.Nom ASC, E.Prenom ASC; -- 6. Nombre d'enseignants dont le grade est 'Moniteur' SELECT COUNT(*) AS Nombre_Moniteurs FROM Enseignant WHERE grade = 'Moniteur'; -- 7. Étudiants n'ayant pas de Fax (valeur NULL) SELECT Nom, Prenom FROM Etudiant WHERE Fax IS NULL; -- 8. Enseignements dont la description contient 'SQL' ou 'Licence' SELECT intitule FROM Enseignement WHERE description LIKE '%SQL%' OR description LIKE '%Licence%'; -- 9. Capacité moyenne et maximum des salles SELECT AVG(capacite) AS Capacite_Moyenne, MAX(capacite) AS Capacite_Max FROM salle; -- 10. Salles dont la capacité est inférieure à la moyenne SELECT batiment, numero_salle, capacite FROM salle WHERE capacite < (SELECT AVG(capacite) FROM salle); -- 11. Classer les étudiants par ville SELECT * FROM Etudiant ORDER BY Ville; -- DDL Inscription & Note CREATE TABLE IF NOT EXISTS Inscription ( Etudiant_ID INTEGER, enseignement_id INTEGER, departement_id INTEGER, PRIMARY KEY (Etudiant_ID, enseignement_id, departement_id), FOREIGN KEY (Etudiant_ID) REFERENCES Etudiant(Etudiant_ID), FOREIGN KEY (enseignement_id, departement_id) REFERENCES enseignement(enseignement_id, departement_id) ); CREATE TABLE IF NOT EXISTS Note ( Note_ID INTEGER PRIMARY KEY AUTO_INCREMENT, Etudiant_ID INTEGER, enseignement_id INTEGER, departement_id INTEGER, valeur_note DECIMAL(4,2) CHECK (valeur_note BETWEEN 0 AND 20), CONSTRAINT FK_Note_Inscription FOREIGN KEY (Etudiant_ID, enseignement_id, departement_id) REFERENCES Inscription(Etudiant_ID, enseignement_id, departement_id) ); -- Trigger DELIMITER // CREATE TRIGGER Verif_Inscription_Note BEFORE INSERT ON Note FOR EACH ROW BEGIN IF NOT EXISTS (SELECT 1 FROM Inscription WHERE Etudiant_ID = NEW.Etudiant_ID ...) THEN INSERT INTO Inscription VALUES (NEW.Etudiant_ID, NEW.enseignement_id, NEW.departement_id); END IF; END; // DELIMITER ;
Script TP 3 — tp3.sql (Oracle)
-- Q1: Creating the three users CREATE USER medali_u1 IDENTIFIED BY pass123; GRANT ALL PRIVILEGES TO medali_u1; CREATE USER medali_u2 IDENTIFIED BY pass456; GRANT CONNECT, RESOURCE TO medali_u2; CREATE USER medali_u3 IDENTIFIED BY pass789; GRANT CONNECT, RESOURCE TO medali_u3; -- Q8: Security Profile creation CREATE PROFILE mon_profil LIMIT FAILED_LOGIN_ATTEMPTS 3 PASSWORD_LIFE_TIME 30; CREATE USER medali_u0 IDENTIFIED BY pass0 PROFILE mon_profil; -- Q9: Delete user DROP USER medali_u0 CASCADE; -- Q3: Create tables CREATE TABLE t1 (id NUMBER PRIMARY KEY, description VARCHAR2(50)); CREATE TABLE t2 (id NUMBER PRIMARY KEY, info VARCHAR2(50)); -- Q4: Rights GRANT SELECT ON t1 TO medali_u1, medali_u3; GRANT SELECT ON t2 TO medali_u1, medali_u3; -- Q6: Revoke REVOKE SELECT ON t1 FROM medali_u1, medali_u3; -- Q10: Role CREATE ROLE r1; GRANT SELECT, INSERT, UPDATE ON t1 TO r1; GRANT r1 TO medali_u1;
